## 스키마 추출
- 인덱스는 테이블이랑 같이 추출된다
```bash
ora2pg -t TABLE    -o tables.sql    -b schema/tables    -c config/ora2pg.conf
ora2pg -t SEQUENCE -o sequences.sql -b schema/sequences -c config/ora2pg.conf
ora2pg -t VIEW     -o views.sql     -b schema/views     -c config/ora2pg.conf
ora2pg -t GRANT    -o grants.sql    -b schema/grants    -c config/ora2pg.conf
```

## oracle과 PostgreSQL에서 확인한다
```bash
# oracle
sqlplus -S scott/tiger@192.168.23.76:1521/orcl <<EOF
SET PAGESIZE 50 LINESIZE 120
SELECT 'EMP'      AS tbl, COUNT(*) cnt FROM emp
UNION ALL SELECT 'DEPT',     COUNT(*) FROM dept
UNION ALL SELECT 'SALGRADE', COUNT(*) FROM salgrade
UNION ALL SELECT 'BONUS',    COUNT(*) FROM bonus
ORDER BY 1;
EXIT
EOF

DEPT              4
EMP              14

# PostgreSQL
psql -c "
SELECT 'EMP'      AS tbl, COUNT(*) FROM scott.emp
UNION ALL SELECT 'DEPT',     COUNT(*) FROM scott.dept
UNION ALL SELECT 'SALGRADE', COUNT(*) FROM scott.salgrade
UNION ALL SELECT 'BONUS',    COUNT(*) FROM scott.bonus
ORDER BY 1;
"
   tbl    | count

 DEPT     |     4
 EMP      |    14

# PostgreSQL 인덱스
psql -c "
SELECT 
    schemaname AS schema,
    tablename AS table,
    indexname AS index_name,
    indexdef AS definition
FROM pg_indexes 
WHERE schemaname = 'scott'
ORDER BY tablename, indexname;
"

# PostgreSQL 뷰
psql -c "
SELECT 
    schemaname AS schema,
    viewname AS view_name,
    definition
FROM pg_views 
WHERE schemaname = 'scott';
"

# PostgreSQL 시퀀스
psql -c "
SELECT 
    n.nspname AS schema,
    c.relname AS sequence_name
FROM pg_class c
JOIN pg_namespace n ON n.oid = c.relnamespace
WHERE n.nspname = 'scott' AND c.relkind = 'S';
"
```
